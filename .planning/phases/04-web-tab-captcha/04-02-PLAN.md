**In offscreen.js -- add `offscreen-poll-captcha` message handler:**

Add a new case in the switch statement (after the existing cases, before `default:`):

```javascript
case 'offscreen-poll-captcha':
    if (!api) {
        sendResponse({ error: 'API not initialized' });
        return true;
    }
    if (!api.jdAPICore || !api.jdAPICore.options.sessiontoken) {
        sendResponse({ error: 'Not logged in' });
        return true;
    }
    // List devices, then check each for pending CAPTCHA jobs
    api.listDevices().done(function(devices) {
        if (!devices || devices.length === 0) {
            sendResponse({ job: null });
            return;
        }
        var found = false;
        var checked = 0;
        devices.forEach(function(device) {
            api.setActiveDevice(device.id);
            api.send('/captcha/getCaptchaJob', '').done(function(captchaData) {
                checked++;
                if (!found && captchaData && captchaData.data && captchaData.data.id) {
                    found = true;
                    // Got a pending job -- fetch full details
                    api.send('/captcha/get', JSON.stringify(captchaData.data.id), 'rawtoken').done(function(jobResponse) {
                        if (jobResponse && jobResponse.data) {
                            sendResponse({
                                job: {
                                    captchaId: captchaData.data.id,
                                    captchaType: captchaData.data.challengeType || captchaData.data.type,
                                    hoster: captchaData.data.hoster,
                                    siteKey: jobResponse.data.siteKey,
                                    siteKeyType: jobResponse.data.type,
                                    v3action: jobResponse.data.v3Action,
                                    targetUrl: jobResponse.data.siteUrl || jobResponse.data.contextUrl,
                                    callbackUrl: 'MYJD'
                                }
                            });
                        } else {
                            sendResponse({ job: null });
                        }
                    }).fail(function() {
                        sendResponse({ job: null });
                    });
                } else if (checked === devices.length && !found) {
                    sendResponse({ job: null });
                }
            }).fail(function() {
                checked++;
                if (checked === devices.length && !found) {
                    sendResponse({ job: null });
                }
            });
        });
    }).fail(function(err) {
        console.error('[Offscreen] CAPTCHA poll failed:', err);
        sendResponse({ job: null, error: err.message || err });
    });
    return true;
```

**IMPORTANT:** The jdapi library uses jQuery Deferred (`.done()/.fail()`), NOT native Promises. Match the existing offscreen.js patterns exactly. The `api.send()` method signature is `api.send(endpoint, params)` for device-scoped calls, and `api.setActiveDevice(deviceId)` must be called first.

**In background.js -- add CAPTCHA polling coordination:**

Add these items in the CAPTCHA section (after `let activeCaptchaTabs = {};`, around line 848):

```javascript
// ============================================================
// CAPTCHA job polling (independent detection without my.jdownloader.org)
// ============================================================
const CAPTCHA_POLL_ALARM = 'captchaPoll';
let processedCaptchaIds = new Set();
let captchaPollingActive = false;

function startCaptchaPolling() {
  if (captchaPollingActive) return;
  captchaPollingActive = true;
  // Chrome alarms minimum period is 30s for packed extensions, lower for unpacked
  chrome.alarms.create(CAPTCHA_POLL_ALARM, { periodInMinutes: 0.5 });
  console.log('Background: CAPTCHA polling started');
}

function stopCaptchaPolling() {
  captchaPollingActive = false;
  chrome.alarms.clear(CAPTCHA_POLL_ALARM);
  processedCaptchaIds.clear();
  console.log('Background: CAPTCHA polling stopped');
}

async function pollForCaptchaJobs() {
  try {
    var result = await sendToOffscreen('offscreen-poll-captcha');
    if (result && result.job && !processedCaptchaIds.has(result.job.captchaId)) {
      processedCaptchaIds.add(result.job.captchaId);
      console.log('Background: CAPTCHA job found via polling:', result.job.hoster, result.job.captchaId);
      handlePolledCaptchaJob(result.job);
    }
  } catch (e) {
    console.error('Background: CAPTCHA poll error:', e);
  }
}

function handlePolledCaptchaJob(job) {
  // Create a new tab (no existing trigger tab when polling finds the job)
  chrome.tabs.create({ url: job.targetUrl + '#rc2jdt', active: true }, function(tab) {
    if (chrome.runtime.lastError) {
      console.error('Background: Failed to create CAPTCHA tab:', chrome.runtime.lastError);
      return;
    }
    chrome.storage.session.set({ myjd_captcha_job: job });
    addCspStrippingRule(tab.id);
    activeCaptchaTabs[tab.id] = {
      callbackUrl: 'MYJD',
      captchaId: job.captchaId,
      captchaType: job.captchaType,
      hoster: job.hoster,
      detectedAt: Date.now()
    };
    console.log('Background: CAPTCHA tab created for polled job:', tab.id, job.hoster);
  });
}
```
